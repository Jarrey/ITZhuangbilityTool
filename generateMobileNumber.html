var values = {{num()}};

function getMobile(values) {
    var mobile = "";
    values.forEach(function(v){
        var s = 0;
        v.forEach(function(i) {
            s = s + i;
        });
        mobile = mobile + s.toString();
    });
    return mobile;
};

alert(getMobile(values));
